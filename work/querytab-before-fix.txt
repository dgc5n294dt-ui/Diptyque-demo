export function QueryTab(): JSX.Element {
  const [question, setQuestion] = useState(SAMPLE_QUESTIONS[0]);
  const [result, setResult] = useState<AskResponse | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState("");

  const providerHint = useMemo(() => result?.provider ?? "mock", [result]);

  async function handleAsk(): Promise<void> {
    if (!question.trim()) return;
    setLoading(true);
    setError("");
    try {
      const response = await askGraphQuestion(question.trim());
      setResult(response);
    } catch (askError) {
      setError(askError instanceof Error ? askError.message : String(askError));
    } finally {
      setLoading(false);
    }
  }

  return (
    <section className="query-tab">
      <div className="section-header">
        <h2>闂瓟</h2>
        <p>褰撳墠浣跨敤 {providerHint} answer provider锛涘悗缁彲鍒囨崲涓?DeepSeek銆?/p>
      </div>

      <div className="query-toolbar">
        <textarea value={question} onChange={(event) => setQuestion(event.target.value)} rows={3} placeholder="杈撳叆涓€涓晢鍝佸浘璋遍棶棰?.." />
        <div className="query-actions">
          <button onClick={() => void handleAsk()} disabled={loading}>{loading ? "妫€绱腑..." : "鍙戦€?}</button>
          <button className="secondary" onClick={() => { setQuestion(""); setResult(null); setError(""); }}>閲嶇疆</button>
        </div>
      </div>

      <div className="sample-question-list">
        {SAMPLE_QUESTIONS.map((sample) => (
          <button key={sample} className="sample-chip" onClick={() => setQuestion(sample)}>{sample}</button>
        ))}
      </div>

      {error ? <div className="error-banner">{error}</div> : null}
      <RetrievalResultView result={result} />
    </section>
  );
}
